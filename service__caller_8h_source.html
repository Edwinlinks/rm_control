52 template <class ServiceType>
53 class ServiceCallerBase
54 {
55 public:
56  explicit ServiceCallerBase(ros::NodeHandle& nh, const std::string& service_name = "") : fail_count_(0), fail_limit_(0)
57  {
58  nh.param("fail_limit", fail_limit_, 0);
59  if (!nh.param("service_name", service_name_, service_name) && service_name.empty())
60  {
61  ROS_ERROR("Service name no defined (namespace: %s)", nh.getNamespace().c_str());
62  return;
63  }
64  client_ = nh.serviceClient<ServiceType>(service_name_);
65  }
66  ServiceCallerBase(XmlRpc::XmlRpcValue& controllers, ros::NodeHandle& nh, const std::string& service_name = "")
67  : fail_count_(0), fail_limit_(0)
68  {
69  if (controllers.hasMember("service_name"))
70  service_name_ = static_cast<std::string>(controllers["service_name"]);
71  else
72  {
73  service_name_ = service_name;
74  if (service_name.empty())
75  {
76  ROS_ERROR("Service name no defined (namespace: %s)", nh.getNamespace().c_str());
77  return;
78  }
79  }
80  client_ = nh.serviceClient<ServiceType>(service_name_);
81  }
82  ~ServiceCallerBase()
83  {
84  delete thread_;
85  }
86  void callService()
87  {
88  if (isCalling())
89  return;
90  thread_ = new std::thread(&ServiceCallerBase::callingThread, this);
91  thread_->detach();
92  }
93  ServiceType& getService()
94  {
95  return service_;
96  }
97  bool isCalling()
98  {
99  std::unique_lock<std::mutex> guard(mutex_, std::try_to_lock);
100  return !guard.owns_lock();
101  }
102 
103 protected:
104  void callingThread()
105  {
106  std::lock_guard<std::mutex> guard(mutex_);
107  if (!client_.call(service_))
108  {
109  ROS_INFO_ONCE("Failed to call service %s on %s. Retrying now ...", typeid(ServiceType).name(),
110  service_name_.c_str());
111  if (fail_limit_ != 0)
112  {
113  fail_count_++;
114  if (fail_count_ >= fail_limit_)
115  {
116  ROS_ERROR_ONCE("Failed to call service %s on %s", typeid(ServiceType).name(), service_name_.c_str());
117  fail_count_ = 0;
118  }
119  }
120  // ros::WallDuration(0.2).sleep();
121  }
122  }
123  std::string service_name_;
124  ros::ServiceClient client_;
125  ServiceType service_;
126  std::thread* thread_{};
127  std::mutex mutex_;
128  int fail_count_, fail_limit_;
129 };
