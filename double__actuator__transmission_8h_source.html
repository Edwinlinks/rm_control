116 DoubleActuatorTransmission::DoubleActuatorTransmission(std::vector<double> actuator_reduction, double joint_reduction,
117  double joint_offset)
118  : act_reduction_(std::move(actuator_reduction)), jnt_reduction_(joint_reduction), jnt_offset_(joint_offset)
119 {
120  if (numActuators() != act_reduction_.size() || numJoints() != 1)
121  {
122  throw TransmissionInterfaceException(
123  "Joint reduction and offset vectors of a double transmission must have size 1, actuator must size 2");
124  }
125  if (0.0 == act_reduction_[0] || 0.0 == act_reduction_[1] || 0.0 == jnt_reduction_)
126  {
127  throw TransmissionInterfaceException("Transmission reduction ratios cannot be zero.");

131 inline void DoubleActuatorTransmission::actuatorToJointEffort(const ActuatorData& act_data, JointData& jnt_data)
132 {
133  assert(numActuators() == act_data.effort.size() && numJoints() == jnt_data.effort.size());
134  assert(act_data.effort[0] && act_data.effort[1] && jnt_data.effort[0]);
135  const std::vector<double>& ar = act_reduction_;
136 
137  *jnt_data.effort[0] = jnt_reduction_ * (*act_data.effort[0] * ar[0] + *act_data.effort[1] * ar[1]);
138 }

140 inline void DoubleActuatorTransmission::actuatorToJointVelocity(const ActuatorData& act_data, JointData& jnt_data)
141 {
142  assert(numActuators() == act_data.velocity.size() && numJoints() == jnt_data.velocity.size());
143  assert(act_data.velocity[0] && act_data.velocity[1] && jnt_data.velocity[0]);
144  const std::vector<double>& ar = act_reduction_;
145 
146  *jnt_data.velocity[0] = (*act_data.velocity[0] / ar[0] + *act_data.velocity[1] / ar[1]) / (2.0 * jnt_reduction_);
147 }

149 inline void DoubleActuatorTransmission::actuatorToJointPosition(const ActuatorData& act_data, JointData& jnt_data)
150 {
151  assert(numActuators() == act_data.position.size() && numJoints() == jnt_data.position.size());
152  assert(act_data.position[0] && act_data.position[1] && jnt_data.position[0]);
153  const std::vector<double>& ar = act_reduction_;
154 
155  // Use first actuator's position
156  *jnt_data.position[0] = (*act_data.position[0] / ar[0]) / jnt_reduction_ + jnt_offset_;
157 }

159 inline void DoubleActuatorTransmission::jointToActuatorEffort(const JointData& jnt_data, ActuatorData& act_data)
160 {
161  assert(numActuators() == act_data.effort.size() && numJoints() == jnt_data.effort.size());
162  assert(act_data.effort[0] && act_data.effort[1] && jnt_data.effort[0]);
163  const std::vector<double>& ar = act_reduction_;
164 
165  // Desired act effort is divided to two actuator evenly.
166  *act_data.effort[0] = (*jnt_data.effort[0] / jnt_reduction_) / (2.0 * ar[0]);
167  *act_data.effort[1] = (*jnt_data.effort[0] / jnt_reduction_) / (2.0 * ar[1]);
168 }
