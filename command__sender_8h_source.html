60 template <class MsgType>
61 class CommandSenderBase
62 {
63 public:
64  explicit CommandSenderBase(ros::NodeHandle& nh)
65  {
66  if (!nh.getParam("topic", topic_))
67  ROS_ERROR("Topic name no defined (namespace: %s)", nh.getNamespace().c_str());
68  queue_size_ = getParam(nh, "queue_size", 1);
69  pub_ = nh.advertise<MsgType>(topic_, queue_size_);
70  }
71  void setMode(int mode)
72  {
73  if (!std::is_same<MsgType, geometry_msgs::Twist>::value && !std::is_same<MsgType, std_msgs::Float64>::value)
74  msg_.mode = mode;
75  }
76  virtual void sendCommand(const ros::Time& time)
77  {
78  pub_.publish(msg_);
79  }
80  virtual void setZero() = 0;
81  MsgType* getMsg()
82  {
83  return &msg_;
84  }
85 
86 protected:
87  std::string topic_;
88  uint32_t queue_size_;
89  ros::Publisher pub_;
90  MsgType msg_;
91 };

194 class GimbalCommandSender : public TimeStampCommandSenderBase<rm_msgs::GimbalCmd>
195 {
196 public:
197  explicit GimbalCommandSender(ros::NodeHandle& nh, const RefereeData& referee_data)
198  : TimeStampCommandSenderBase<rm_msgs::GimbalCmd>(nh, referee_data)
199  {
200  if (!nh.getParam("max_yaw_vel", max_yaw_rate_))
201  ROS_ERROR("Max yaw velocity no defined (namespace: %s)", nh.getNamespace().c_str());
202  if (!nh.getParam("max_pitch_vel", max_pitch_vel_))
203  ROS_ERROR("Max pitch velocity no defined (namespace: %s)", nh.getNamespace().c_str());
204  if (!nh.getParam("track_timeout", track_timeout_))
205  ROS_ERROR("Track timeout no defined (namespace: %s)", nh.getNamespace().c_str());
206  if (!nh.getParam("eject_sensitivity", eject_sensitivity_))
207  eject_sensitivity_ = 1.;
208  cost_function_ = new TargetCostFunction(nh, referee_data);
209  }
210  ~GimbalCommandSender()
211  {
212  delete cost_function_;
213  };
214  void setRate(double scale_yaw, double scale_pitch)
215  {
216  msg_.rate_yaw = scale_yaw * max_yaw_rate_;
217  msg_.rate_pitch = scale_pitch * max_pitch_vel_;
218  if (eject_flag_)
219  {
220  msg_.rate_yaw *= eject_sensitivity_;
221  msg_.rate_pitch *= eject_sensitivity_;
222  }
223  }
224  void setZero() override
225  {
226  msg_.rate_yaw = 0.;
227  msg_.rate_pitch = 0.;
228  }
229  void setBulletSpeed(double bullet_speed)
230  {
231  msg_.bullet_speed = bullet_speed;
232  }
233  void updateCost(const rm_msgs::TrackDataArray& track_data_array)
234  {
235  double cost = cost_function_->costFunction(track_data_array);
236  if (!track_data_array.tracks.empty())
237  last_track_ = ros::Time::now();
238  if (cost == 1e9 && (ros::Time::now() - last_track_).toSec() > track_timeout_)
239  setMode(rm_msgs::GimbalCmd::RATE);
240  else
241  setMode(rm_msgs::GimbalCmd::TRACK);
242  }
243  void setEject(bool flag)
244  {
245  eject_flag_ = flag;
246  }
247  bool getEject() const
248  {
249  return eject_flag_;
250  }
251 
252 private:
253  TargetCostFunction* cost_function_;
254  ros::Time last_track_;
255  double max_yaw_rate_{}, max_pitch_vel_{}, track_timeout_{}, eject_sensitivity_ = 1.;
256  bool eject_flag_{};
257 };

266  heat_limit_ = new HeatLimit(limit_nh, referee_data);
267  if (!nh.getParam("gimbal_error_limit", gimbal_error_limit_))
268  ROS_ERROR("gimbal error limit no defined (namespace: %s)", nh.getNamespace().c_str());
269  }
270  ~ShooterCommandSender()
271  {
272  delete heat_limit_;
273  }
274  void checkError(const rm_msgs::GimbalDesError& gimbal_des_error, const ros::Time& time)
275  {
276  if (gimbal_des_error.error > gimbal_error_limit_ && time - gimbal_des_error.stamp < ros::Duration(0.1))
277  if (msg_.mode == rm_msgs::ShootCmd::PUSH)
278  setMode(rm_msgs::ShootCmd::READY);
279  }
280  void sendCommand(const ros::Time& time) override
281  {
282  msg_.speed = heat_limit_->getSpeedLimit();
283  msg_.hz = heat_limit_->getHz();
284  TimeStampCommandSenderBase<rm_msgs::ShootCmd>::sendCommand(time);
285  }
286  double getSpeed()
287  {
288  switch (msg_.speed)
289  {
290  case rm_msgs::ShootCmd::SPEED_10M_PER_SECOND:
291  return 10.;
292  case rm_msgs::ShootCmd::SPEED_15M_PER_SECOND:
293  return 15.;
294  case rm_msgs::ShootCmd::SPEED_16M_PER_SECOND:
295  return 16.;
296  case rm_msgs::ShootCmd::SPEED_18M_PER_SECOND:
297  return 18.;
298  case rm_msgs::ShootCmd::SPEED_30M_PER_SECOND:
299  return 30.;
300  }
301  return 0.;
302  }
303  void setBurstMode(bool burst_flag)
304  {
305  heat_limit_->setMode(burst_flag);
306  }
307  bool getBurstMode()
308  {
309  return heat_limit_->getMode();
310  }
311  void setZero() override{};
312 
313 private:
314  double gimbal_error_limit_{};
315  HeatLimit* heat_limit_{};
