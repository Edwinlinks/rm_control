68  double getHz() const
69  {
70  if (burst_flag_)
71  return shoot_frequency_;
72  if (!referee_data_.is_online_)
73  return safe_shoot_frequency_;
74  double cooling_limit{}, cooling_rate{}, cooling_heat{};
75  if (type_ == "ID1_17MM")
76  {
77  cooling_limit = referee_data_.game_robot_status_.shooter_id_1_17_mm_cooling_limit_;
78  cooling_rate = referee_data_.game_robot_status_.shooter_id_1_17_mm_cooling_rate_;
79  cooling_heat = referee_data_.power_heat_data_.shooter_id_1_17_mm_cooling_heat_;
80  }
81  else if (type_ == "ID2_17MM")
82  {
83  cooling_limit = referee_data_.game_robot_status_.shooter_id_2_17_mm_cooling_limit_;
84  cooling_rate = referee_data_.game_robot_status_.shooter_id_2_17_mm_cooling_rate_;
85  cooling_heat = referee_data_.power_heat_data_.shooter_id_2_17_mm_cooling_heat_;
86  }
87  else if (type_ == "ID1_42MM")
88  {
89  cooling_limit = referee_data_.game_robot_status_.shooter_id_1_42_mm_cooling_limit_;
90  cooling_rate = referee_data_.game_robot_status_.shooter_id_1_42_mm_cooling_rate_;
91  cooling_heat = referee_data_.power_heat_data_.shooter_id_1_42_mm_cooling_heat_;
92  }
93 
94  if (cooling_limit - cooling_heat < bullet_heat_)
95  return 0.0;
96  else if (cooling_limit - cooling_heat == bullet_heat_)
97  return cooling_rate / bullet_heat_;
98  else if (cooling_limit - cooling_heat <= bullet_heat_ * heat_coeff_)
99  return (cooling_limit - cooling_heat) / (bullet_heat_ * heat_coeff_) *
100  (shoot_frequency_ - cooling_rate / bullet_heat_) +
101  cooling_rate / bullet_heat_;
102  else
103  return shoot_frequency_;
104  }
