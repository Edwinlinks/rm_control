79  double getLimitPower()
80  {
81  if (referee_data_.robot_id_ == rm_common::RobotId::BLUE_SENTRY ||
82  referee_data_.robot_id_ == rm_common::RobotId::RED_SENTRY)
83  limit_power_ = 30;
84  else if (referee_data_.robot_id_ == rm_common::RobotId::RED_ENGINEER ||
85  referee_data_.robot_id_ == rm_common::RobotId::BLUE_ENGINEER)
86  limit_power_ = 300;
87  else
88  { // standard and hero
89  if (referee_data_.is_online_)
90  {
91  if (referee_data_.capacity_data.is_online_)
92  {
93  if (referee_data_.game_status_.game_progress_ == 1)
94  return 30; // calibra
95  if (referee_data_.game_robot_status_.chassis_power_limit_ > 120)
96  limit_power_ = burst_power_;
97  else
98  {
99  switch (state_)
100  {
101  case TEST:
102  test();
103  break;
104  case BURST:
105  burst();
106  break;
107  case NORMAL:
108  normal();
109  break;
110  case CHARGE:
111  charge();
112  break;
113  }
114  if (!(state_ == Mode::BURST) && (abs(referee_data_.capacity_data.limit_power_ -
115  referee_data_.game_robot_status_.chassis_power_limit_) < 0.05))
116  normal();
117  }
118  }
119  else
120  limit_power_ = referee_data_.game_robot_status_.chassis_power_limit_;
121  }
122  else
123  limit_power_ = safety_power_;
124  }
125  return limit_power_;
126  }
