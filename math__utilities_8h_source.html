42 T angularMinus(T a, T b)
43 {
44  a = fmod(a, 2.0 * M_PI);
45  b = fmod(b, 2.0 * M_PI);
46 
47  T res1 = a - b;
48  T res2 = (a < b) ? (a + 2 * M_PI - b) : (a - 2 * M_PI - b);
49 
50  return (std::abs(res1) < std::abs(res2)) ? res1 : res2;
51 }
